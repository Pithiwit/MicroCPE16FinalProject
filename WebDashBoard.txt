    <script>
        let liveSensorRGB = { r: 0, g: 0, b: 0 };
        // ไฟล์แยกใช้ wss ได้ปกติ (เพราะไม่ได้ครอบด้วย IP http)
        const client = mqtt.connect('wss://broker.hivemq.com:8884/mqtt');
        
        // *** แก้ไข: รับข้อมูลจากช่องทางใหม่ (_v2) ***
        const topic_sub = "cpe16/spectramatch/sensor_v2"; 

        const ctx = document.getElementById('rgbChart').getContext('2d');
        const rgbChart = new Chart(ctx, {
            type: 'bar',
            data: { labels: ['Red', 'Green', 'Blue'], datasets: [{ data: [0, 0, 0], backgroundColor: ['#ff4757', '#2ed573', '#1e90ff'], borderWidth: 0, borderRadius: 5 }] },
            options: { scales: { y: { beginAtZero: true, max: 255, grid: { color: '#333' } }, x: { grid: { display: false } } }, plugins: { legend: { display: false } }, animation: { duration: 100 } }
        });

        client.on('connect', () => {
            document.getElementById('mqttStatus').innerHTML = "🟢 <b>เชื่อมต่อ MQTT สำเร็จแล้ว!</b>";
            client.subscribe(topic_sub);
        });

        client.on('message', (topic, message) => {
            if(topic === topic_sub) {
                try {
                    const data = JSON.parse(message.toString());
                    liveSensorRGB = { r: data.r, g: data.g, b: data.b };
                    
                    document.getElementById('rgbText').innerText = `R: ${data.r} | G: ${data.g} | B: ${data.b}`;
                    document.getElementById('colorBox').style.backgroundColor = `rgb(${data.r}, ${data.g}, ${data.b})`;
                    document.getElementById('colorBox').style.boxShadow = `0 0 20px rgb(${data.r}, ${data.g}, ${data.b})`;
                    rgbChart.data.datasets[0].data = [data.r, data.g, data.b];
                    rgbChart.update();

                    checkMatch(); 
                } catch (e) {
                    console.log("ข้ามข้อความที่ไม่ใช่ JSON:", message.toString());
                }
            }
        });

        function checkMatch() {
            const hexGuess = document.getElementById('guessPicker').value;
            const guessR = parseInt(hexGuess.substr(1, 2), 16);
            const guessG = parseInt(hexGuess.substr(3, 2), 16);
            const guessB = parseInt(hexGuess.substr(5, 2), 16);
            
            const actual = liveSensorRGB;

            document.getElementById('revealTitle').style.color = "#fff";
            document.getElementById('resultBox').style.backgroundColor = `rgb(${actual.r}, ${actual.g}, ${actual.b})`;
            document.getElementById('resultBox').style.boxShadow = `0 0 15px rgb(${actual.r}, ${actual.g}, ${actual.b})`;

            const distance = Math.sqrt(Math.pow(guessR - actual.r, 2) + Math.pow(guessG - actual.g, 2) + Math.pow(guessB - actual.b, 2));
            const maxDistance = 441.67;
            let matchPercentage = ((1 - (distance / maxDistance)) * 100).toFixed(2);
            if (matchPercentage < 0) matchPercentage = 0;
            
            const scoreText = document.getElementById('scoreText');
            scoreText.innerText = matchPercentage + "%";
            const feedback = document.getElementById('gameFeedback');
            
            if (matchPercentage >= 90) {
                scoreText.style.color = "#2ed573"; scoreText.style.textShadow = "0 0 20px rgba(46, 213, 115, 0.6)";
                feedback.innerHTML = "🎯 เพอร์เฟกต์! แม่นเหมือนจับวาง"; feedback.style.color = "#2ed573";
            } else if (matchPercentage >= 75) {
                scoreText.style.color = "#1e90ff"; scoreText.style.textShadow = "0 0 20px rgba(30, 144, 255, 0.6)";
                feedback.innerHTML = "👍 ใกล้เคียงมาก โทนสีมาถูกทางแล้ว"; feedback.style.color = "#1e90ff";
            } else {
                scoreText.style.color = "#ff4757"; scoreText.style.textShadow = "0 0 20px rgba(255, 71, 87, 0.6)";
                feedback.innerHTML = "😅 ยังไม่ค่อยตรง ลองปรับเฉดสีใหม่ดูนะ"; feedback.style.color = "#ff4757";
            }
        }
